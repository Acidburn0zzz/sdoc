    def build_indices
      @toplevels.each do |toplevel|
        @files << SHtmlFile.new(toplevel, @options, FILE_DIR)
      end
      @topclasses = []
      RDoc::TopLevel.all_classes_and_modules.each do |cls|
        @topclasses << build_class_list(cls, @files[0], CLASS_DIR)
      end
    end
