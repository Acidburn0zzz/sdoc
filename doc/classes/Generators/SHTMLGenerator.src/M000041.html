    def gen_tree_level(classes)
      tree = []
      classes.select{|c| c.content? }.sort.each do |item|
        item = [item.title, item.namespace, item.path_if_available, item.params, item.snippet, gen_tree_level(item.children)]
        tree << item
      end
      tree
    end
