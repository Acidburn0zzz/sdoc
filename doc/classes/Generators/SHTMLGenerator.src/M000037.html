    def build_class_list(from, html_file, class_dir)
      klass = SHtmlClass.new(from, html_file, class_dir, @options)
      @classes << klass
      from.each_classmodule do |mod|
        klass.children << build_class_list(mod, html_file, class_dir)
      end
      klass
    end
