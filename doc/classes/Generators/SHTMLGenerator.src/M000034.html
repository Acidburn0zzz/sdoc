    def load_html_template
      template = @options.template
      unless template =~ %r{/|\\}
        template = File.join("sdoc/generators/template",
                             @options.generator.key, template)
      end
      require template
      extend RDoc::Page
    rescue LoadError
      $stderr.puts "Could not find HTML template '#{template}'"
      exit 99
    end
